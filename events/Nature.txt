province_event = { #Start Blight
	id = YW_nature.2
	title = YW_nature.2.t
	desc = YW_nature.2.d
	picture = FAMINE_eventPicture

	trigger = {
		OR = {
			trade_goods = grain
			trade_goods = cocoa
			trade_goods = cotton
			trade_goods = tobacco
		}
		NOT = { has_province_modifier = crop_failure }
		OR = {
			has_winter = mild_winter
			has_winter = normal_winter
		}
	}
	
	mean_time_to_happen = {
		months = 2400
		modifier = {
			factor = 0.75
			any_neighbor_province = {
				is_empty = yes
			}
		}
	}

	immediate =  {
		add_province_modifier = { 
			name = blight_modifier
			duration = 1825
		}
		hidden_effect = {
			set_variable = {
				which = blightCost
				value = 5
			}
			owner = {
				every_owned_province = {
					limit = {
						OR = {
							trade_goods = grain
							trade_goods = cocoa
							trade_goods = cotton
							trade_goods = tobacco
						}
					}
					ROOT = {
						change_variable = {
							which = blightCost
							value = 2.5
						}
					}
				}
			}
		}
	}

	option = {
		name = YW_nature.2.a
		if = {
			limit = { 
				NOT = { check_variable = { which = blightCost value = 10 } }
			}
			owner = {
				add_treasury = -10
			}
		}
		else_if = {
			limit = { 
				check_variable = { which = blightCost value = 10 }
				NOT = { check_variable = { which = blightCost value = 25 } }
			}
			owner = {
				add_treasury = -25
			}
		}
		else_if = {
			limit = { 
				check_variable = { which = blightCost value = 25 }
				NOT = { check_variable = { which = blightCost value = 50 } }
			}
			owner = {
				add_treasury = -50
			}
		}
		else_if = {
			limit = { 
				check_variable = { which = blightCost value = 50 }
				NOT = { check_variable = { which = blightCost value = 100 } }
			}
			owner = {
				add_treasury = -100
			}
		}
		else_if = {
			limit = { 
				check_variable = { which = blightCost value = 100 }
			}
			owner = {
				add_treasury = -250
			}
		}
		else = {
			log = "NATURE: [Root.blightCost.GetValue] is not an accepted value"
		}
		owner = {
			set_country_flag = blight_relief
		}
	}
	option = {
		name = YW_nature.2.c
		add_province_modifier = { 
			name = burned_crops
			duration = 1825
		}
		random = {
			chance = 75
			hidden_effect = {
				remove_province_modifier = blight_modifier
			}
		}
	}
}
